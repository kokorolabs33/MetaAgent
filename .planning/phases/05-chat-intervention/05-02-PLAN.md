if (event.type === "agent.typing") {
  const data = event.data as Record<string, unknown>;
  const agentId = data.agent_id as string;
  const agentName = data.agent_name as string;
  if (agentId && agentName) {
    set((s) => ({
      typingAgents: { ...s.typingAgents, [agentId]: agentName },
    }));
    // Client-side safety timeout: auto-clear after 65s (slightly longer than backend 60s)
    // Prevents stuck indicators if agent.typing_stopped event is lost (Pitfall 5)
    setTimeout(() => {
      set((s) => {
        if (s.typingAgents[agentId]) {
          const { [agentId]: _, ...rest } = s.typingAgents;
          return { typingAgents: rest };
        }
        return s;
      });
    }, 65000);
  }
}

if (event.type === "agent.typing_stopped") {
  const data = event.data as Record<string, unknown>;
  const agentId = data.agent_id as string;
  if (agentId) {
    set((s) => {
      const { [agentId]: _, ...rest } = s.typingAgents;
      return { typingAgents: rest };
    });
  }
}

sendMessage: async (taskId, content) => {
  const result = await api.messages.send(taskId, content);
  // Handle wrapped response with advisory_errors (D-02)
  const msg = (result as Record<string, unknown>).message
    ? ((result as Record<string, unknown>).message as Message)
    : (result as Message);
  const advisoryErrors = (result as Record<string, unknown>).advisory_errors as string[] | undefined;

  // Optimistically add message — SSE dedup will skip if it arrives again
  set((s) => {
    if (s.messages.some((m) => m.id === msg.id)) return s;
    return { messages: [...s.messages, msg] };
  });

    // Store advisory errors as a transient notification — caller handles display

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| SSE event stream -> Zustand store | Backend SSE events (agent.typing, message with metadata) are processed by store; malformed events must not crash the UI |
| Store -> React components | typingAgents map and message metadata drive UI rendering; stale data must auto-clear |
| User input -> autocomplete selection | User selects agents from autocomplete; inactive agent selection must be blocked client-side |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-07 | Denial of Service | typingAgents store state | mitigate | 65-second client-side safety timeout auto-clears stale typing indicators; agent.typing_stopped SSE event provides normal cleanup |
| T-05-08 | Tampering | SSE event data | accept | SSE events come from authenticated backend over same-origin connection; event data is typed before use. Malformed data results in no-op (missing fields checked before state update) |
| T-05-09 | Spoofing | advisory_errors response | accept | Errors originate from authenticated backend response to user's own request; displayed as informational text only, no HTML rendering |
| T-05-10 | Information Disclosure | metadata in SSE message event | accept | Advisory metadata (advisory: true, subtask_id) is non-sensitive; only visible to the user who created the task and is viewing the conversation |
</threat_model>
