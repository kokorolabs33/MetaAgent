    ```typescript
    /**
     * connectMultiTaskSSE subscribes to the multiplexed task stream at
     * /api/tasks/stream?ids=a,b,c. All events for the listed task IDs arrive
     * on a single EventSource connection, keeping the dashboard under the
     * browser's 6-connection-per-domain HTTP/1.1 limit (per Phase 4 INTR-03).
     *
     * Returns a disconnect function that calls EventSource.close().
     * If taskIds is empty, returns a no-op disconnect.
     */
    export function connectMultiTaskSSE(
      taskIds: string[],
      onEvent: SSEEventHandler,
      onError?: (error: Event) => void,
    ): () => void {
      if (taskIds.length === 0) {
        return () => {};
      }
      const url = `${BASE}/api/tasks/stream?ids=${encodeURIComponent(taskIds.join(","))}`;
      const eventSource = new EventSource(url, { withCredentials: true });

      eventSource.onmessage = (e: MessageEvent) => {
        try {
          const parsed = JSON.parse(e.data as string) as Record<string, unknown>;
          onEvent({
            id: e.lastEventId,
            type: parsed.type as string,
            data: (parsed.data as Record<string, unknown>) ?? {},
            ...(parsed.subtask_id ? { subtask_id: parsed.subtask_id as string } : {}),
            ...(parsed.task_id ? { task_id: parsed.task_id as string } : {}),
            ...(parsed.actor_type ? { actor_type: parsed.actor_type as string } : {}),
            ...(parsed.actor_id ? { actor_id: parsed.actor_id as string } : {}),
            ...(parsed.created_at ? { created_at: parsed.created_at as string } : {}),
          });
        } catch {
          // ignore malformed events
        }
      };

      eventSource.onerror = (e: Event) => {
        if (onError) onError(e);
        // EventSource auto-reconnects with Last-Event-ID header
      };

      return () => {
        eventSource.close();
      };
    }
    ```

    // ─── Dashboard Store ────────────────────────────────────────

    type TaskProgress = { completed: number; total: number };

    interface DashboardStore {
      tasks: Task[];
      totalTasks: number;
      currentPage: number;
      totalPages: number;
      taskProgress: Record<string, TaskProgress>;
      isLoading: boolean;
      error: string | null;
      multiTaskSSEDisconnect: (() => void) | null;

      loadDashboard: (params?: { status?: string; q?: string; page?: number }) => Promise<void>;
      connectDashboardSSE: (taskIds: string[]) => void;
      disconnectDashboardSSE: () => void;
      handleDashboardEvent: (event: TaskEvent) => void;
    }

    export const useDashboardStore = create<DashboardStore>((set, get) => ({
      tasks: [],
      totalTasks: 0,
      currentPage: 1,
      totalPages: 0,
      taskProgress: {},
      isLoading: false,
      error: null,
      multiTaskSSEDisconnect: null,

      loadDashboard: async (params) => {
        set({ isLoading: true, error: null });
        try {
          const result = await api.tasks.list({
            ...params,
            per_page: 12, // UI-SPEC: 12 per page (4 rows × 3 cols)
          });
          // Build initial progress map from list response (Plan 01 extended
          // the query to return completed_subtasks + total_subtasks).
          const progress: Record<string, TaskProgress> = {};
          for (const t of result.items) {
            progress[t.id] = {
              completed: t.completed_subtasks,
              total: t.total_subtasks,
            };
          }
          set({
            tasks: result.items,
            totalTasks: result.total,
            currentPage: result.page,
            totalPages: result.pages,
            taskProgress: progress,
            isLoading: false,
          });
        } catch (err) {
          console.error("loadDashboard failed:", err);
          set({ isLoading: false, error: "Failed to load tasks. Check your connection and try again." });
        }
      },

      connectDashboardSSE: (taskIds) => {
        get().disconnectDashboardSSE();
        if (taskIds.length === 0) return;
        const disconnect = connectMultiTaskSSE(taskIds, (event) => {
          get().handleDashboardEvent(event as unknown as TaskEvent);
        });
        set({ multiTaskSSEDisconnect: disconnect });
      },

      disconnectDashboardSSE: () => {
        const disconnect = get().multiTaskSSEDisconnect;
        if (disconnect) {
          disconnect();
          set({ multiTaskSSEDisconnect: null });
        }
      },

      handleDashboardEvent: (event) => {
        const taskId = event.task_id;
        if (!taskId) return;

        // Task-level status updates: patch the task in the list.
        if (event.type.startsWith("task.")) {
          const statusMap: Record<string, Task["status"]> = {
            "task.planning": "planning",
            "task.running": "running",
            "task.completed": "completed",
            "task.failed": "failed",
            "task.cancelled": "cancelled",
            "task.approval_required": "approval_required",
          };
          const newStatus = statusMap[event.type];
          if (newStatus) {
            set((s) => ({
              tasks: s.tasks.map((t) =>
                t.id === taskId ? { ...t, status: newStatus } : t,
              ),
            }));
          }
          return;
        }

        // Subtask-level updates: maintain the progress map deltas.
        // Per research recommendation 4: subtask.completed AND subtask.failed
        // both count toward the progress bar's "completed" denominator because
        // they are both finalized states.
        if (event.type === "subtask.created") {
          set((s) => {
            const cur = s.taskProgress[taskId] ?? { completed: 0, total: 0 };
            return {
              taskProgress: {
                ...s.taskProgress,
                [taskId]: { completed: cur.completed, total: cur.total + 1 },
              },
            };
          });
          return;
        }
        if (event.type === "subtask.completed" || event.type === "subtask.failed") {
          set((s) => {
            const cur = s.taskProgress[taskId];
            if (!cur) return s;
            // Cap completed at total to avoid visual > 100% if SSE outruns list.
            const nextCompleted = Math.min(cur.completed + 1, cur.total);
            return {
              taskProgress: {
                ...s.taskProgress,
                [taskId]: { completed: nextCompleted, total: cur.total },
              },
            };
          });
          return;
        }
      },
    }));

    "use client";

    import { useEffect, useMemo } from "react";
    import { useRouter, useSearchParams, usePathname } from "next/navigation";
    import { Loader2, AlertCircle } from "lucide-react";
    import { useDashboardStore, useAgentStore } from "@/lib/store";
    import { TaskFilterBar } from "@/components/dashboard/TaskFilterBar";
    import { DashboardTaskCard } from "@/components/dashboard/DashboardTaskCard";
    import {
      DashboardEmptyState,
      type DashboardEmptyStateVariant,
    } from "@/components/dashboard/DashboardEmptyState";
    import { Pagination } from "@/components/ui/pagination";

    export function TaskDashboard() {
      const searchParams = useSearchParams();
      const pathname = usePathname();
      const router = useRouter();

      const status = searchParams.get("status") ?? undefined;
      const q = searchParams.get("q") ?? undefined;
      const page = Math.max(1, Number(searchParams.get("page") ?? "1"));

      const tasks = useDashboardStore((s) => s.tasks);
      const totalPages = useDashboardStore((s) => s.totalPages);
      const currentPage = useDashboardStore((s) => s.currentPage);
      const isLoading = useDashboardStore((s) => s.isLoading);
      const error = useDashboardStore((s) => s.error);
      const loadDashboard = useDashboardStore((s) => s.loadDashboard);
      const connectDashboardSSE = useDashboardStore((s) => s.connectDashboardSSE);
      const disconnectDashboardSSE = useDashboardStore((s) => s.disconnectDashboardSSE);
      const taskProgress = useDashboardStore((s) => s.taskProgress);

      const connectStatusSSE = useAgentStore((s) => s.connectStatusSSE);
      const disconnectStatusSSE = useAgentStore((s) => s.disconnectStatusSSE);
      const getAgentStatus = useAgentStore((s) => s.getAgentStatus);

      // Load on filter change. useEffect dep is the primitive strings, not
      // objects, so changes to the store itself do not re-trigger the load.
      useEffect(() => {
        void loadDashboard({ status, q, page });
      }, [status, q, page, loadDashboard]);

      // Connect multiplexed task SSE keyed to the visible task IDs.
      // Use ids.join(",") as the effect dep (Pitfall 4 in research) so a new
      // array reference with the same ids does NOT tear down the connection.
      const idsKey = useMemo(() => tasks.map((t) => t.id).join(","), [tasks]);
      useEffect(() => {
        if (idsKey === "") {
          disconnectDashboardSSE();
          return;
        }
        connectDashboardSSE(idsKey.split(","));
        return () => disconnectDashboardSSE();
      }, [idsKey, connectDashboardSSE, disconnectDashboardSSE]);

      // Connect global agent status stream for dots.
      useEffect(() => {
        connectStatusSSE();
        return () => disconnectStatusSSE();
      }, [connectStatusSSE, disconnectStatusSSE]);

      const handlePageChange = (nextPage: number) => {
        const params = new URLSearchParams(searchParams.toString());
        if (nextPage <= 1) {
          params.delete("page");
        } else {
          params.set("page", String(nextPage));
        }
        const qs = params.toString();
        router.replace(qs ? `${pathname}?${qs}` : pathname);
      };

      // Determine empty-state variant.
      const emptyVariant: DashboardEmptyStateVariant = q
        ? "search"
        : status === "running"
          ? "running"
          : status === "completed"
            ? "completed"
            : status === "failed"
              ? "failed"
              : "all";

      // Agent IDs per card: we don't have a per-task agent list in the Task
      // interface today (subtasks carry agent_id, but Plan 01 did not add an
      // agent summary to the list response). For v1, pass an empty array —
      // the dots column renders blank, and the UI remains informative. This
      // is acceptable per UI-SPEC (dots are a supplementary signal, not the
      // primary data). A future enhancement can add an agent-ids aggregate
      // to the list query, paralleling the subtask counts.
      const agentIdsForTask = (_taskId: string): string[] => [];

      return (
        <div
          className="flex h-full flex-col"
          aria-live="polite"
        >
          <TaskFilterBar />

          <div className="flex-1 overflow-auto px-6 py-6">
            {error ? (
              <div className="flex h-full flex-col items-center justify-center px-4 text-center">
                <AlertCircle
                  className="mb-4 size-10 text-destructive"
                  aria-hidden="true"
                />
                <p className="text-sm text-muted-foreground">{error}</p>
              </div>
            ) : isLoading && tasks.length === 0 ? (
              <div className="flex h-full items-center justify-center">
                <Loader2
                  className="size-6 animate-spin text-muted-foreground"
                  aria-label="Loading tasks"
                />
              </div>
            ) : tasks.length === 0 ? (
              <DashboardEmptyState variant={emptyVariant} />
            ) : (
              <>
                <div className="grid grid-cols-1 gap-4 sm:grid-cols-2 lg:grid-cols-3">
                  {tasks.map((task) => (
                    <DashboardTaskCard
                      key={task.id}
                      task={task}
                      progress={taskProgress[task.id]}
                      agentIds={agentIdsForTask(task.id)}
                      getAgentStatus={getAgentStatus}
                    />
                  ))}
                </div>
                {totalPages > 1 && (
                  <div className="mt-6 flex justify-center">
                    <Pagination
                      currentPage={currentPage}
                      totalPages={totalPages}
                      onPageChange={handlePageChange}
                    />
                  </div>
                )}
              </>
            )}
          </div>
        </div>
      );

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → EventSource /api/tasks/stream | Untrusted URL built from tasks array; auth is enforced server-side by middleware (Plan 01 registered the route inside r.Group authed) |
| URL query params → TaskDashboard + TaskFilterBar | Untrusted status/q/page values read via useSearchParams; status is whitelisted against API values, q is URL-encoded, page is coerced to number + clamped via Math.max(1, ...) |
| SSE event payload → useDashboardStore.handleDashboardEvent | Events from broker originate from trusted executor; still JSON-parsed with try/catch at the SSE boundary (existing sse.ts pattern); handler only reads known fields |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-13 | Tampering | page query param `page` | mitigate | `Math.max(1, Number(...))` clamps to at least 1; the backend per_page cap of 100 from tasks.go:102 already bounds page size; invalid values fall through to page=1 with no handler error. |
| T-04-14 | Denial of service | SSE reconnect loop | mitigate | `idsKey` memo uses `tasks.map(...).join(",")` so the useEffect only re-runs when the actual id set changes. Reference changes alone do not trigger disconnect/reconnect churn (research Pitfall 4). |
| T-04-15 | Information disclosure | Error message display | mitigate | Error string is a hardcoded user-facing message ("Failed to load tasks. Check your connection and try again.") — backend error details are logged via console.error but NOT rendered, preventing stack-trace leaks to a shoulder-surfer. |
| T-04-16 | Spoofing | SSE event origination | accept | Events arrive via authenticated EventSource connection; the browser sends session cookies via `withCredentials: true`. The server side (Plan 01) subscribes the same broker topics the per-task stream uses, so the trust model matches the existing per-task observability pattern. |
| T-04-17 | Tampering | URL search params fed to api.tasks.list | mitigate | `api.tasks.list` builds a URLSearchParams object (web/lib/api.ts:83-88) which URL-encodes all values. The backend (Plan 01) parameterises SQL via pgx. No injection path. |
